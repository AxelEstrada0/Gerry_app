class CounterPage extends StatefulWidget {
  const CounterPage({super.key});

  @override
  State<CounterPage> createState() => _CounterPageState();
}

class _CounterPageState extends State<CounterPage> {
  int _contador = 0;

  Future<void> _cargarContador() async {
    final prefs = await SharedPreferences.getInstance();
    setState(() {
      _contador = prefs.getInt('contador') ?? 0;
    });
  }

  Future<void> _guardarContador() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setInt('contador', _contador);
  }

  @override
  void initState() {
    super.initState();
    _cargarContador();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Nivel 6 - Memoria'),
      ),
      body: Center(
        child: Column(
          mainAxisSize: MainAxisSize.min,
          children: [
            Text(
              'Contador: $_contador',
              style: const TextStyle(fontSize: 32),
            ),
            const SizedBox(height: 20),
            ElevatedButton(
              onPressed: () {
                setState(() {
                  _contador++;
                });
                _guardarContador();
              },
              child: const Text('Sumar'),
            ),
            ElevatedButton(
              onPressed: () {
                Navigator.pop(context);
              },
              child: const Text('Volver'),
            ),
          ],
        ),
      ),
    );
  }

}

class MemoryListPage extends StatefulWidget {
  const MemoryListPage({super.key});

  @override
  State<MemoryListPage> createState() => _MemoryListPageState();
}

class _MemoryListPageState extends State<MemoryListPage> {
  final TextEditingController _controller = TextEditingController();
  List<String> _items = [];

  // ===== MEMORIA =====

  Future<void> _cargarLista() async {
    final prefs = await SharedPreferences.getInstance();
    final String? data = prefs.getString('lista');

    if (data != null) {
      final List decoded = jsonDecode(data);
      setState(() {
        _items = decoded.cast<String>();
      });
    }
  }

  Future<void> _guardarLista() async {
    final prefs = await SharedPreferences.getInstance();
    final String encoded = jsonEncode(_items);
    await prefs.setString('lista', encoded);
  }

  // ===== CICLO DE VIDA =====

  @override
  void initState() {
    super.initState();
    _cargarLista();
  }

  // ===== UI =====

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Nivel 7 - Lista viva'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [

            // INPUT
            TextField(
              controller: _controller,
              decoration: const InputDecoration(
                labelText: 'Nuevo item',
                border: OutlineInputBorder(),
              ),
            ),

            const SizedBox(height: 12),

            // BOTÓN AGREGAR
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: () {
                  if (_controller.text.trim().isEmpty) return;

                  setState(() {
                    _items.add(_controller.text.trim());
                    _controller.clear();
                  });

                  _guardarLista();
                },
                child: const Text('Agregar'),
              ),
            ),

            const SizedBox(height: 16),

            // LISTA
            Expanded(
              child: _items.isEmpty
                  ? const Center(child: Text('Lista vacía 🧠'))
                  : ListView.builder(
                      itemCount: _items.length,
                      itemBuilder: (context, index) {
                        return Card(
                          child: ListTile(
                            title: Text(_items[index]),
                            trailing: IconButton(
                              icon: const Icon(Icons.delete),
                              onPressed: () {
                                setState(() {
                                  _items.removeAt(index);
                                });
                                _guardarLista();
                              },
                            ),
                          ),
                        );
                      },
                    ),
            ),

            // VOLVER
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: [
                SizedBox(
                  // width: double.infinity,
                  child: ElevatedButton(
                    onPressed: () {
                      Navigator.pop(context);
                    },
                    child: const Text('Volver'),
                  ),
                ),
                ElevatedButton(
                  onPressed: () {
                    Navigator.push(
                      context,
                      MaterialPageRoute(
                        builder: (context) => const NumberListPage(),
                      ),
                    );
                  },
                  child: const Text('Int List'),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }
}

class NumberListPage extends StatefulWidget {
  const NumberListPage({super.key});

  @override
  State<NumberListPage> createState() => _NumberListPageState();
}

class _NumberListPageState extends State<NumberListPage> {

  // ===== CONTROLADOR =====
  final TextEditingController _numController = TextEditingController();

  // ===== DATA =====
  List<int> _numeros = [];

  // ===== MEMORIA =====
  Future<void> _cargarLista() async {
    final prefs = await SharedPreferences.getInstance();
    final String? data = prefs.getString('lista_numeros');

    if (data != null) {
      final List decoded = jsonDecode(data);
      setState(() {
        _numeros = decoded.cast<int>();
      });
    }
  }

  Future<void> _guardarLista() async {
    final prefs = await SharedPreferences.getInstance();
    final String encoded = jsonEncode(_numeros);
    await prefs.setString('lista_numeros', encoded);
  }

  // ===== CICLO DE VIDA =====
  @override
  void initState() {
    super.initState();
    _cargarLista();
  }

  // ===== UI =====
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Nivel 7 - Lista viva'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [

            // INPUT
            TextField(
              controller: _numController,
              keyboardType: TextInputType.number,
              decoration: const InputDecoration(
                labelText: 'Agregar número',
                border: OutlineInputBorder(),
              ),
            ),

            const SizedBox(height: 12),

            // BOTÓN AGREGAR
            ElevatedButton(
              onPressed: () {
                if (_numController.text.trim().isEmpty) return;

                final int? valor = int.tryParse(_numController.text.trim());
                if (valor == null) return;

                setState(() {
                  _numeros.add(valor);
                });

                _guardarLista();
                _numController.clear();
              },
              child: const Text('Agregar número'),
            ),

            const SizedBox(height: 16),

            // LISTA
            Expanded(
              child: ListView.builder(
                itemCount: _numeros.length,
                itemBuilder: (context, index) {
                  return Card(
                    child: ListTile(
                      title: Text(
                        _numeros[index].toString(),
                        style: const TextStyle(fontSize: 18),
                      ),
                      trailing: IconButton(
                        icon: const Icon(Icons.delete),
                        onPressed: () {
                          setState(() {
                            _numeros.removeAt(index);
                          });
                          _guardarLista();
                        },
                      ),
                    ),
                  );
                },
              ),
            ),

            // VOLVER
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: () {
                  Navigator.pop(context);
                },
                child: const Text('Volver'),
              ),
            ),
          ],
        ),
      ),
    );
  }
}

class CloudPage extends StatefulWidget {
  const CloudPage({super.key});

  @override
  State<CloudPage> createState() => _CloudPageState();
}

class _CloudPageState extends State<CloudPage> {
  final TextEditingController _controller = TextEditingController();
  final FirebaseFirestore _db = FirebaseFirestore.instance;

  // ===== GUARDAR EN LA NUBE =====

  Future<void> _guardarTexto() async {

    final texto = _controller.text.trim();
    
    if (texto.isEmpty) return;

    await _db.collection('mensajes').add({
      'texto': texto,
      'timestamp': FieldValue.serverTimestamp(),
    });

    _controller.clear();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Nivel 8 - Cloud'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [

            // INPUT
            TextField(
              controller: _controller,
              decoration: const InputDecoration(
                labelText: 'Mensaje a la nube',
                border: OutlineInputBorder(),
              ),
            ),

            const SizedBox(height: 12),

            // BOTÓN GUARDAR
            ElevatedButton(
              onPressed: _guardarTexto,
              child: const Text('Guardar en la nube'),
            ),

            const SizedBox(height: 20),

            // LISTA EN TIEMPO REAL
            Expanded(
              child: StreamBuilder<QuerySnapshot>(
                stream: _db
                    .collection('mensajes')
                    .orderBy('timestamp', descending: true)
                    .snapshots(),
                builder: (context, snapshot) {
                  if (snapshot.connectionState == ConnectionState.waiting) {
                    return const Center(child: CircularProgressIndicator());
                  }

                  if (!snapshot.hasData || snapshot.data!.docs.isEmpty) {
                    return const Center(child: Text('Sin datos aún'));
                  }

                  final docs = snapshot.data!.docs;

                  return ListView.builder(
                    itemCount: docs.length,
                    itemBuilder: (context, index) {
                      final data = docs[index].data() as Map<String, dynamic>;
                      return Card(
                        child: ListTile(
                          title: Text(data['texto'] ?? ''),
                        ),
                      );
                    },
                  );
                },
              ),
            ),

            // VOLVER
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: () {
                  Navigator.pop(context);
                },
                child: const Text('Volver'),
              ),
            ),
          ],
        ),
      ),
    );
  }
}
